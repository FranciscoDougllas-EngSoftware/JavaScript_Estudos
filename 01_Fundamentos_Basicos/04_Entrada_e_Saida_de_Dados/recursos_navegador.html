Recursos do Navegador

console.log("nome"); - Imprime no console

console.clear(); - Limpa o console

console.error("mensagem"); - Imprime um erro no console

console.warn("mensagem"); - Imprime um aviso no console

console.time("nome"); - Inicia um contador

console.timeEnd("nome"); - Finaliza um contador

console.table(["item1", "item2", "item3"]); - Imprime uma tabela no console

console.dir(objeto); - Imprime um objeto no console

window.alert("mensagem"); - Imprime uma mensagem no navegador

window.confirm("mensagem"); - Imprime uma caixa de confirmação

window.prompt("mensagem"); - Imprime uma caixa de entrada
